#include <stdio.h>
#include <math.h>

int main() 
{
    int year;
    printf("Enter the year :\n");
    scanf("%d",&year);
     
    if(year%4==0 && year%100!=0)
     {
     printf("The year is a leap year");
     }
    else
    {
     printf("The year is NOT a leap year");
    }
   return 0;
}
