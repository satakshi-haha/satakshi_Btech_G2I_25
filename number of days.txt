// Online C compiler to run C program online
#include <stdio.h>

int main()
{
    int days, years, months, rem;
    printf("Enter the days:\n");
    scanf("%d", &days);
    years= days/365;
    months=(days%365)/30;
    rem=(days%365)-(months*30);
    printf("%dyears, %dmonths, %ddays", years, months, rem);
    return 0;
}
